    function digPow(num, pow) {
  const raisedAndSummed = num.toString()
                             .split('')
                             .map(n => Math.pow(n, pow++))
                             .reduce((sum, n) => sum + n, 0)
  return (raisedAndSummed % num === 0) ? raisedAndSummed / num : -1
}
